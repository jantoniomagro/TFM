\section{NoSQL}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection{Definición}
\begin{frame}
\frametitle{NoSQL}

NoSQL = \emph{Not Only SQL}

\begin{multicols}{2}

\textbf{Ventajas}
\begin{itemize}
\item Escalabilidad
\item Modelos de datos flexibles
\item Bajo coste
\end{itemize}

\textbf{Inconvenientes}
\begin{itemize}
\item Relativamente reciente
\item Soporte
\item Pocos usuarios
\end{itemize}

\end{multicols}

\end{frame}
